let numbers = [1; 2; 3; 4; 5; 6]
let evenSquares = numbers |> List.filter (fun n -> n % 2 = 0) |> List.map (fun n -> n * n)
printfn "Even squares: %A" evenSquares
let fibonacciSeq = seq { let mutable a = 0; let mutable b = 1; while true do yield a; let next = a + b; a <- b; b <- next }
printfn "First 10 Fibonacci: %A" (fibonacciSeq |> Seq.take 10 |> Seq.toList)
