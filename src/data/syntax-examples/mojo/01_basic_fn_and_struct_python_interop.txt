fn mojo_greet(name: String, loud: Bool = False):

struct Point: var x: Float64; var y: Float64
    fn __init__(inout self, x: Float64, y: Float64): self.x=x; self.y=y
    fn norm(self) -> Float64: return (self.x**2 + self.y**2).sqrt()
